296
297int main(
298 int argc,
299 char *argv[]) {
300
301 ctl_t ctl;
302
303 atm_t *atm;
304
305 cache_t *cache;
306
307 clim_t *clim;
308
309 met_t *met0, *met1;
310
311#ifdef ASYNCIO
312 met_t *met0TMP, *met1TMP, *mets;
313 ctl_t ctlTMP;
314#endif
315
316 FILE *dirlist;
317
318 char dirname[LEN], filename[2 * LEN];
319
320 double *dt, *rs, t;
321
322 int ntask = -1, rank = 0, size = 1;
323
324 /* Start timers... */
325 START_TIMERS;
326
327 /* Initialize MPI... */
328#ifdef MPI
329 SELECT_TIMER("MPI_INIT", "INIT", NVTX_CPU);
330 MPI_Init(&argc, &argv);
331 MPI_Comm_rank(MPI_COMM_WORLD, &rank);
332 MPI_Comm_size(MPI_COMM_WORLD, &size);
333#endif
334
335 /* Initialize GPUs... */
336#ifdef _OPENACC
337 SELECT_TIMER("ACC_INIT", "INIT", NVTX_GPU);
338 if (acc_get_num_devices(acc_device_nvidia) <= 0)
339 ERRMSG("Not running on a GPU device!");
340 acc_set_device_num(rank % acc_get_num_devices(acc_device_nvidia),
341 acc_device_nvidia);
342 acc_device_t device_type = acc_get_device_type();
343 acc_init(device_type);
344#endif
345
346 /* Check arguments... */
347 if (argc < 4)
348 ERRMSG("Give parameters: <dirlist> <ctl> <atm_in>");
349
350 /* Open directory list... */
351 if (!(dirlist = fopen(argv[1], "r")))
352 ERRMSG("Cannot open directory list!");
353
354 /* Loop over directories... */
355 while (fscanf(dirlist, "%4999s", dirname) != EOF) {
356
357 /* MPI parallelization... */
358 if ((++ntask) % size != rank)
359 continue;
360
361 /* ------------------------------------------------------------
362 Initialize model run...
363 ------------------------------------------------------------ */
364
365 /* Allocate... */
366 SELECT_TIMER("ALLOC", "MEMORY", NVTX_CPU);
367 ALLOC(atm, atm_t, 1);
368 ALLOC(cache, cache_t, 1);
369 ALLOC(clim, clim_t, 1);
370 ALLOC(met0, met_t, 1);
371 ALLOC(met1, met_t, 1);
372#ifdef ASYNCIO
373 ALLOC(met0TMP, met_t, 1);
374 ALLOC(met1TMP, met_t, 1);
375#endif
376 ALLOC(dt, double,
377 NP);
378 ALLOC(rs, double,
379 3 * NP + 1);
380
381 /* Create data region on GPUs... */
382#ifdef _OPENACC
383 SELECT_TIMER("CREATE_DATA_REGION", "MEMORY", NVTX_GPU);
384#ifdef ASYNCIO
385#pragma acc enter data create(atm[:1], cache[:1], clim[:1], ctl, ctlTMP, met0[:1], met1[:1], met0TMP[:1], met1TMP[:1], dt[:NP], rs[:3 * NP])
386#else
387#pragma acc enter data create(atm[:1], cache[:1], clim[:1], ctl, met0[:1], met1[:1], dt[:NP], rs[:3 * NP])
388#endif
389#endif
390
391 /* Read control parameters... */
392 sprintf(filename, "%s/%s", dirname, argv[2]);
393 read_ctl(filename, argc, argv, &ctl);
394
395 /* Read climatological data... */
396 read_clim(&ctl, clim);
397
398 /* Read atmospheric data... */
399 sprintf(filename, "%s/%s", dirname, argv[3]);
400 if (!read_atm(filename, &ctl, atm))
401 ERRMSG("Cannot open file!");
402
403 /* Initialize timesteps... */
404 module_timesteps_init(&ctl, atm);
405
406 /* Update GPU... */
407#ifdef _OPENACC
408 SELECT_TIMER("UPDATE_DEVICE", "MEMORY", NVTX_H2D);
409#pragma acc update device(atm[:1], clim[:1], ctl)
410#endif
411
412 /* Initialize random number generator... */
413 module_rng_init(ntask);
414
415 /* Initialize meteo data... */
416#ifdef ASYNCIO
417 ctlTMP = ctl;
418#endif
419 get_met(&ctl, clim, ctl.t_start, &met0, &met1);
420 if (ctl.dt_mod > fabs(met0->lon[1] - met0->lon[0]) * 111132. / 150.)
421 WARN("Violation of CFL criterion! Check DT_MOD!");
422#ifdef ASYNCIO
423 get_met(&ctlTMP, clim, ctlTMP.t_start, &met0TMP, &met1TMP);
424#endif
425
426 /* Initialize isosurface... */
427 if (ctl.isosurf >= 1 && ctl.isosurf <= 4)
428 module_isosurf_init(&ctl, met0, met1, atm, cache);
429
430 /* Initialize pressure heights consistent with zeta ... */
431 if (ctl.vert_coord_ap == 1) {
432 INTPOL_INIT;
433 for (int ip = 0; ip < atm->np; ip++) {
434 intpol_met_4d_coord(met0, met0->zetal, met0->pl, met1, met1->zetal,
435 met1->pl, atm->time[ip], atm->q[ctl.qnt_zeta][ip],
436 atm->lon[ip], atm->lat[ip], &atm->p[ip], ci, cw,
437 1);
438 }
439 }
440
441 /* Update GPU... */
442#ifdef _OPENACC
443 SELECT_TIMER("UPDATE_DEVICE", "MEMORY", NVTX_H2D);
444#pragma acc update device(cache[:1])
445#endif
446
447 /* ------------------------------------------------------------
448 Loop over timesteps...
449 ------------------------------------------------------------ */
450
451 /* Loop over timesteps... */
452#ifdef ASYNCIO
453 omp_set_nested(1);
454 // omp_set_dynamic(0);
455 int ompTrdnum = omp_get_max_threads();
456#endif
457 for (t = ctl.t_start; ctl.direction * (t - ctl.t_stop) < ctl.dt_mod;
458 t += ctl.direction * ctl.dt_mod) {
459#ifdef ASYNCIO
460#pragma omp parallel num_threads(2)
461 {
462#endif
463
464 /* Adjust length of final time step... */
465 if (ctl.direction * (t - ctl.t_stop) > 0)
466 t = ctl.t_stop;
467
468 /* Set time steps of air parcels... */
469 module_timesteps(&ctl, met0, atm, dt, t);
470
471 /* Get meteo data... */
472#ifdef ASYNCIO
473#pragma acc wait(5)
474#pragma omp barrier
475 if (omp_get_thread_num() == 0) {
476
477 /* Pointer swap... */
478 if (t != ctl.t_start) {
479 mets = met0;
480 met0 = met0TMP;
481 met0TMP = mets;
482
483 mets = met1;
484 met1 = met1TMP;
485 met1TMP = mets;
486 }
487#endif
488#ifndef ASYNCIO
489 if (t != ctl.t_start)
490 get_met(&ctl, clim, t, &met0, &met1);
491#endif
492
493 /* Sort particles... */
494 if (ctl.sort_dt > 0 && fmod(t, ctl.sort_dt) == 0)
495 module_sort(&ctl, met0, atm);
496
497 /* Check positions (initial)... */
498 module_position(&ctl, met0, met1, atm, dt);
499
500 /* Advection... */
501 if (ctl.advect > 0) {
502 if (ctl.vert_coord_ap == 0)
503 module_advect(&ctl, met0, met1, atm, dt);
504 else
505 module_advect_diabatic(&ctl, met0, met1, atm, dt);
506 }
507
508 /* Turbulent diffusion... */
509 if (ctl.turb_dx_trop > 0 || ctl.turb_dz_trop > 0
510 || ctl.turb_dx_strat > 0 || ctl.turb_dz_strat > 0)
511 module_diffusion_turb(&ctl, clim, atm, dt, rs);
512
513 /* Mesoscale diffusion... */
514 if (ctl.turb_mesox > 0 || ctl.turb_mesoz > 0)
515 module_diffusion_meso(&ctl, met0, met1, atm, cache, dt, rs);
516
517 /* Convection... */
518 if (ctl.conv_cape >= 0
519 && (ctl.conv_dt <= 0 || fmod(t, ctl.conv_dt) == 0))
520 module_convection(&ctl, met0, met1, atm, dt, rs);
521
522 /* Sedimentation... */
523 if (ctl.qnt_rp >= 0 && ctl.qnt_rhop >= 0)
524 module_sedi(&ctl, met0, met1, atm, dt);
525
526 /* Isosurface... */
527 if (ctl.isosurf >= 1 && ctl.isosurf <= 4)
528 module_isosurf(&ctl, met0, met1, atm, cache, dt);
529
530 /* Check positions (final)... */
531 module_position(&ctl, met0, met1, atm, dt);
532
533 /* Interpolate meteo data... */
534 if (ctl.met_dt_out > 0
535 && (ctl.met_dt_out < ctl.dt_mod
536 || fmod(t, ctl.met_dt_out) == 0))
537 module_meteo(&ctl, clim, met0, met1, atm, dt);
538
539 /* Check boundary conditions (initial)... */
540 if ((ctl.bound_lat0 < ctl.bound_lat1)
541 && (ctl.bound_p0 > ctl.bound_p1))
542 module_bound_cond(&ctl, clim, met0, met1, atm, dt);
543
544 /* Decay of particle mass... */
545 if (ctl.tdec_trop > 0 && ctl.tdec_strat > 0)
546 module_decay(&ctl, clim, atm, dt);
547
548 /* Interparcel mixing... */
549 if (ctl.mixing_trop >= 0 && ctl.mixing_strat >= 0
550 && (ctl.mixing_dt <= 0 || fmod(t, ctl.mixing_dt) == 0))
551 module_mixing(&ctl, clim, atm, t);
552
553 /* OH chemistry... */
554 if (ctl.oh_chem_reaction != 0)
555 module_oh_chem(&ctl, clim, met0, met1, atm, dt);
556
557 /* H2O2 chemistry (for SO2 aqueous phase oxidation)... */
558 if (ctl.h2o2_chem_reaction != 0) {
559 module_chemgrid(&ctl, clim, met0, met1, atm, t);
560 module_h2o2_chem(&ctl, clim, met0, met1, atm, dt);
561 }
562
563 /* First-order tracer chemistry... */
564 if (ctl.tracer_chem)
565 module_tracer_chem(&ctl, clim, met0, met1, atm, dt);
566
567 /* KPP chemistry... */
568 if (ctl.kpp_chem) {
569#ifdef KPP
570 module_chemgrid(&ctl, clim, met0, met1, atm, t);
571 module_kpp_chem(&ctl, clim, met0, met1, atm, dt);
572#else
573 ERRMSG("Code was compiled without KPP!");
574#endif
575 }
576
577 /* Wet deposition... */
578 if ((ctl.wet_depo_ic_a > 0 || ctl.wet_depo_ic_h[0] > 0)
579 && (ctl.wet_depo_bc_a > 0 || ctl.wet_depo_bc_h[0] > 0))
580 module_wet_deposition(&ctl, met0, met1, atm, dt);
581
582 /* Dry deposition... */
583 if (ctl.dry_depo_vdep > 0)
584 module_dry_deposition(&ctl, met0, met1, atm, dt);
585
586 /* Check boundary conditions (final)... */
587 if ((ctl.bound_lat0 < ctl.bound_lat1)
588 && (ctl.bound_p0 > ctl.bound_p1))
589 module_bound_cond(&ctl, clim, met0, met1, atm, dt);
590
591 /* Write output... */
592 write_output(dirname, &ctl, met0, met1, atm, t);
593#ifdef ASYNCIO
594 } else {
595 omp_set_num_threads(ompTrdnum);
596 if (ctl.direction * (t - ctl.t_stop + ctl.direction * ctl.dt_mod) <
597 ctl.dt_mod)
598 get_met(&ctl, clim, t + (ctl.direction * ctl.dt_mod), &met0TMP,
599 &met1TMP);
600 }
601 }
602#endif
603 }
604
605#ifdef ASYNCIO
606 omp_set_num_threads(ompTrdnum);
607#endif
608
609 /* ------------------------------------------------------------
610 Finalize model run...
611 ------------------------------------------------------------ */
612
613 /* Report problem size... */
614 LOG(1, "SIZE_NP = %d", atm->np);
615 LOG(1, "SIZE_MPI_TASKS = %d", size);
616 LOG(1, "SIZE_OMP_THREADS = %d", omp_get_max_threads());
617#ifdef _OPENACC
618 LOG(1, "SIZE_ACC_DEVICES = %d", acc_get_num_devices(acc_device_nvidia));
619#endif
620
621 /* Report memory usage... */
622 LOG(1, "MEMORY_ATM = %g MByte", sizeof(atm_t) / 1024. / 1024.);
623 LOG(1, "MEMORY_CACHE = %g MByte", sizeof(cache_t) / 1024. / 1024.);
624 LOG(1, "MEMORY_CLIM = %g MByte", sizeof(clim_t) / 1024. / 1024.);
625 LOG(1, "MEMORY_METEO = %g MByte", 2 * sizeof(met_t) / 1024. / 1024.);
626 LOG(1, "MEMORY_DYNAMIC = %g MByte", (3 * NP * sizeof(int)
627 + 4 * NP * sizeof(double)
628 + EX * EY * EP * sizeof(float)) /
629 1024. / 1024.);
630 LOG(1, "MEMORY_STATIC = %g MByte", (EX * EY * EP * sizeof(float)) /
631 1024. / 1024.);
632
633 /* Delete data region on GPUs... */
634#ifdef _OPENACC
635 SELECT_TIMER("DELETE_DATA_REGION", "MEMORY", NVTX_GPU);
636#ifdef ASYNCIO
637#pragma acc exit data delete (ctl, atm, cache, clim, met0, met1, dt, rs, met0TMP, met1TMP)
638#else
639#pragma acc exit data delete (ctl, atm, cache, clim, met0, met1, dt, rs)
640#endif
641#endif
642
643 /* Free... */
644 SELECT_TIMER("FREE", "MEMORY", NVTX_CPU);
645 free(atm);
646 free(cache);
647 free(clim);
648 free(met0);
649 free(met1);
650#ifdef ASYNCIO
651 free(met0TMP);
652 free(met1TMP);
653#endif
654 free(dt);
655 free(rs);
656
657 /* Report timers... */
658 PRINT_TIMERS;
659 }
660
661 /* Finalize MPI... */
662#ifdef MPI
663 MPI_Finalize();
664#endif
665
666 /* Stop timers... */
667 STOP_TIMERS;
668
669 return EXIT_SUCCESS;
670}

672/*****************************************************************************/
673
674void module_advect(
675 ctl_t * ctl,
676 met_t * met0,
677 met_t * met1,
678 atm_t * atm,
679 double *dt) {
680
681 /* Set timer... */
682 SELECT_TIMER("MODULE_ADVECTION", "PHYSICS", NVTX_GPU);
683
684 /* Loop over particles... */
685 PARTICLE_LOOP(0, atm->np, 1, "acc data present(ctl,met0,met1,atm,dt)") {
686
687 /* Init... */
688 double dts, u[4], um = 0, v[4], vm = 0, w[4], wm = 0, x[3] = { 0, 0, 0 };
689
690 /* Loop over integration nodes... */
691 for (int i = 0; i < ctl->advect; i++) {
692
693 /* Set position... */
694 if (i == 0) {
695 dts = 0.0;
696 x[0] = atm->lon[ip];
697 x[1] = atm->lat[ip];
698 x[2] = atm->p[ip];
699 } else {
700 dts = (i == 3 ? 1.0 : 0.5) * dt[ip];
701 x[0] = atm->lon[ip] + DX2DEG(dts * u[i - 1] / 1000., atm->lat[ip]);
702 x[1] = atm->lat[ip] + DY2DEG(dts * v[i - 1] / 1000.);
703 x[2] = atm->p[ip] + dts * w[i - 1];
704 }
705 double tm = atm->time[ip] + dts;
706
707 /* Interpolate meteo data... */
708#ifdef UVW
709 intpol_met_time_uvw(met0, met1, tm, x[2], x[0], x[1],
710 &u[i], &v[i], &w[i]);
711#else
712 INTPOL_INIT;
713 intpol_met_time_3d(met0, met0->u, met1, met1->u, tm,
714 x[2], x[0], x[1], &u[i], ci, cw, 1);
715 intpol_met_time_3d(met0, met0->v, met1, met1->v, tm,
716 x[2], x[0], x[1], &v[i], ci, cw, 0);
717 intpol_met_time_3d(met0, met0->w, met1, met1->w, tm,
718 x[2], x[0], x[1], &w[i], ci, cw, 0);
719#endif
720
721 /* Get mean wind... */
722 double k = 1.0;
723 if (ctl->advect == 2)
724 k = (i == 0 ? 0.0 : 1.0);
725 else if (ctl->advect == 4)
726 k = (i == 0 || i == 3 ? 1.0 / 6.0 : 2.0 / 6.0);
727 um += k * u[i];
728 vm += k * v[i];
729 wm += k * w[i];
730 }
731
732 /* Set new position... */
733 atm->time[ip] += dt[ip];
734 atm->lon[ip] += DX2DEG(dt[ip] * um / 1000.,
735 (ctl->advect == 2 ? x[1] : atm->lat[ip]));
736 atm->lat[ip] += DY2DEG(dt[ip] * vm / 1000.);
737 atm->p[ip] += dt[ip] * wm;
738 }
739}

744 ctl_t * ctl,
745 met_t * met0,

747 atm_t * atm,
748 double *dt) {
749
750 /* Set timer... */
751 SELECT_TIMER("MODULE_ADVECTION", "PHYSICS", NVTX_GPU);
752
753 /* Loop over particles... */
754 PARTICLE_LOOP(0, atm->np, 1, "acc data present(ctl,met0,met1,atm,dt)") {
755
756 /* If other modules have changed p translate it into a zeta... */
757 if (ctl->cpl_zeta_and_press_modules > 0) {
758 INTPOL_INIT;
759 intpol_met_4d_coord(met0, met0->pl, met0->zetal, met1,
760 met1->pl, met1->zetal, atm->time[ip], atm->p[ip],
761 atm->lon[ip], atm->lat[ip],
762 &atm->q[ctl->qnt_zeta][ip], ci, cw, 1);
763 }
764
765 /* Init... */
766 double dts, u[4], um = 0, v[4], vm = 0, zeta_dot[4], zeta_dotm = 0,
767 x[3] = { 0, 0, 0 };
768
769 /* Loop over integration nodes... */
770 for (int i = 0; i < ctl->advect; i++) {
771
772 /* Set position... */
773 if (i == 0) {
774 dts = 0.0;
775 x[0] = atm->lon[ip];
776 x[1] = atm->lat[ip];
777 x[2] = atm->q[ctl->qnt_zeta][ip];
778 } else {
779 dts = (i == 3 ? 1.0 : 0.5) * dt[ip];
780 x[0] = atm->lon[ip] + DX2DEG(dts * u[i - 1] / 1000., atm->lat[ip]);
781 x[1] = atm->lat[ip] + DY2DEG(dts * v[i - 1] / 1000.);
782 x[2] = atm->q[ctl->qnt_zeta][ip] + dts * zeta_dot[i - 1];
783 }
784 double tm = atm->time[ip] + dts;
785
786 /* Interpolate meteo data... */
787 INTPOL_INIT;
788 intpol_met_4d_coord(met0, met0->zetal, met0->ul, met1, met1->zetal,
789 met1->ul, tm, x[2], x[0], x[1], &u[i], ci, cw, 1);
790 intpol_met_4d_coord(met0, met0->zetal, met0->vl, met1, met0->zetal,
791 met1->vl, tm, x[2], x[0], x[1], &v[i], ci, cw, 0);
792 intpol_met_4d_coord(met0, met0->zetal, met0->zeta_dotl, met1,
793 met1->zetal, met1->zeta_dotl, tm, x[2], x[0], x[1],
794 &zeta_dot[i], ci, cw, 0);
795
796 /* Get mean wind... */
797 double k = 1.0;
798 if (ctl->advect == 2)
799 k = (i == 0 ? 0.0 : 1.0);
800 else if (ctl->advect == 4)
801 k = (i == 0 || i == 3 ? 1.0 / 6.0 : 2.0 / 6.0);
802 um += k * u[i];
803 vm += k * v[i];
804 zeta_dotm += k * zeta_dot[i];
805 }
806
807 /* Set new position... */
808 atm->time[ip] += dt[ip];
809 atm->lon[ip] += DX2DEG(dt[ip] * um / 1000.,
810 (ctl->advect == 2 ? x[1] : atm->lat[ip]));
811 atm->lat[ip] += DY2DEG(dt[ip] * vm / 1000.);
812 atm->q[ctl->qnt_zeta][ip] += dt[ip] * zeta_dotm;
813
814 /* Check if zeta is below zero... */
815 if (atm->q[ctl->qnt_zeta][ip] < 0) {
816 atm->q[ctl->qnt_zeta][ip] = 0;
817 }
818
819 /* Set new position also in pressure coordinates... */
820 INTPOL_INIT;
821 intpol_met_4d_coord(met0, met0->zetal, met0->pl, met1, met1->zetal,
822 met1->pl, atm->time[ip], atm->q[ctl->qnt_zeta][ip],
823 atm->lon[ip], atm->lat[ip], &atm->p[ip], ci, cw, 1);
824 }

937
938 /* Loop over particles... */
939 PARTICLE_LOOP(0, atm->np, 1, "acc data present(ctl,met0,met1,atm,dt,rs)") {

942 double cape;

944 INTPOL_2D(cape, 1);

955 }
956
957 /* Interpolate equilibrium level... */
958 double pel;
959 INTPOL_2D(pel, 0);
960
961 /* Check whether particle is above cloud top... */
962 if (!isfinite(pel) || atm->p[ip] < pel)

965 /* Set pressure range for vertical mixing... */
966 double pbot = atm->p[ip];
967 double ptop = atm->p[ip];

969 double ps;
970 INTPOL_2D(ps, 0);
971 pbot = ps;
972 }
973 if (ctl->conv_mix_top == 1)
974 ptop = pel;
975

977 double tbot, ttop;
978 intpol_met_time_3d(met0, met0->t, met1, met1->t, atm->time[ip],

981 ptop, atm->lon[ip], atm->lat[ip], &ttop, ci, cw, 1);

983 double rhotop = ptop / ttop;
984

986 double rho = rhobot + (rhotop - rhobot) * rs[ip];

988 /* Get pressure... */
989 atm->p[ip] = LIN(rhobot, pbot, rhotop, ptop, rho);

1033void module_diffusion_meso(
1034 ctl_t * ctl,
1035 met_t * met0,
1036 met_t * met1,
1037 atm_t * atm,
1038 cache_t * cache,
1039 double *dt,
1040 double *rs) {
1041
1042 /* Set timer... */
1043 SELECT_TIMER("MODULE_TURBMESO", "PHYSICS", NVTX_GPU);
1044
1045 /* Create random numbers... */
1046 module_rng(rs, 3 * (size_t) atm->np, 1);
1047
1048 /* Loop over particles... */
1049 PARTICLE_LOOP(0, atm->np, 1,
1050 "acc data present(ctl,met0,met1,atm,cache,dt,rs)") {
1051
1052 /* Get indices... */
1053 int ix = locate_reg(met0->lon, met0->nx, atm->lon[ip]);
1054 int iy = locate_reg(met0->lat, met0->ny, atm->lat[ip]);
1055 int iz = locate_irr(met0->p, met0->np, atm->p[ip]);
1056
1057 /* Get standard deviations of local wind data... */
1058 float umean = 0, usig = 0, vmean = 0, vsig = 0, wmean = 0, wsig = 0;
1059 for (int i = 0; i < 2; i++)
1060 for (int j = 0; j < 2; j++)
1061 for (int k = 0; k < 2; k++) {
1062#ifdef UVW
1063 umean += met0->uvw[ix + i][iy + j][iz + k][0];
1064 usig += SQR(met0->uvw[ix + i][iy + j][iz + k][0]);
1065 vmean += met0->uvw[ix + i][iy + j][iz + k][1];
1066 vsig += SQR(met0->uvw[ix + i][iy + j][iz + k][1]);
1067 wmean += met0->uvw[ix + i][iy + j][iz + k][2];
1068 wsig += SQR(met0->uvw[ix + i][iy + j][iz + k][2]);
1069
1070 umean += met1->uvw[ix + i][iy + j][iz + k][0];
1071 usig += SQR(met1->uvw[ix + i][iy + j][iz + k][0]);
1072 vmean += met1->uvw[ix + i][iy + j][iz + k][1];
1073 vsig += SQR(met1->uvw[ix + i][iy + j][iz + k][1]);
1074 wmean += met1->uvw[ix + i][iy + j][iz + k][2];
1075 wsig += SQR(met1->uvw[ix + i][iy + j][iz + k][2]);
1076#else
1077 umean += met0->u[ix + i][iy + j][iz + k];
1078 usig += SQR(met0->u[ix + i][iy + j][iz + k]);
1079 vmean += met0->v[ix + i][iy + j][iz + k];
1080 vsig += SQR(met0->v[ix + i][iy + j][iz + k]);
1081 wmean += met0->w[ix + i][iy + j][iz + k];
1082 wsig += SQR(met0->w[ix + i][iy + j][iz + k]);
1083
1084 umean += met1->u[ix + i][iy + j][iz + k];
1085 usig += SQR(met1->u[ix + i][iy + j][iz + k]);
1086 vmean += met1->v[ix + i][iy + j][iz + k];
1087 vsig += SQR(met1->v[ix + i][iy + j][iz + k]);
1088 wmean += met1->w[ix + i][iy + j][iz + k];
1089 wsig += SQR(met1->w[ix + i][iy + j][iz + k]);
1090#endif
1091 }
1092 usig = usig / 16.f - SQR(umean / 16.f);
1093 usig = (usig > 0 ? sqrtf(usig) : 0);
1094 vsig = vsig / 16.f - SQR(vmean / 16.f);
1095 vsig = (vsig > 0 ? sqrtf(vsig) : 0);
1096 wsig = wsig / 16.f - SQR(wmean / 16.f);
1097 wsig = (wsig > 0 ? sqrtf(wsig) : 0);
1098
1099 /* Set temporal correlations for mesoscale fluctuations... */
1100 double r = 1 - 2 * fabs(dt[ip]) / ctl->dt_met;
1101 double r2 = sqrt(1 - r * r);
1102
1103 /* Calculate horizontal mesoscale wind fluctuations... */
1104 if (ctl->turb_mesox > 0) {
1105 cache->uvwp[ip][0] =
1106 (float) (r * cache->uvwp[ip][0] +
1107 r2 * rs[3 * ip] * ctl->turb_mesox * usig);
1108 atm->lon[ip] +=
1109 DX2DEG(cache->uvwp[ip][0] * dt[ip] / 1000., atm->lat[ip]);
1110
1111 cache->uvwp[ip][1] =
1112 (float) (r * cache->uvwp[ip][1] +
1113 r2 * rs[3 * ip + 1] * ctl->turb_mesox * vsig);
1114 atm->lat[ip] += DY2DEG(cache->uvwp[ip][1] * dt[ip] / 1000.);
1115 }
1116
1117 /* Calculate vertical mesoscale wind fluctuations... */
1118 if (ctl->turb_mesoz > 0) {
1119 cache->uvwp[ip][2] =
1120 (float) (r * cache->uvwp[ip][2] +
1121 r2 * rs[3 * ip + 2] * ctl->turb_mesoz * wsig);
1122 atm->p[ip] += cache->uvwp[ip][2] * dt[ip];
1123 }
1124 }
1125}

1157 }
1158
1159 /* Vertical turbulent diffusion... */
1160 if (dz > 0) {
1161 double sigma = sqrt(2.0 * dz * fabs(dt[ip]));
1162 atm->p[ip] += DZ2DP(rs[3 * ip + 2] * sigma / 1000., atm->p[ip]);
1163 }
1164 }

1502 || ixs[ip] < 0 || ixs[ip] >= ctl->chemgrid_nx

1619void module_oh_chem(
1620 ctl_t * ctl,
1621 clim_t * clim,
1622 met_t * met0,
1623 met_t * met1,
1624 atm_t * atm,
1625 double *dt) {
1626
1627 /* Set timer... */
1628 SELECT_TIMER("MODULE_OHCHEM", "PHYSICS", NVTX_GPU);
1629
1630 /* Check quantity flags... */
1631 if (ctl->qnt_m < 0 && ctl->qnt_vmr < 0)
1632 ERRMSG("Module needs quantity mass or volume mixing ratio!");
1633
1634 /* Loop over particles... */
1635 PARTICLE_LOOP(0, atm->np, 1, "acc data present(ctl,clim,met0,met1,atm,dt)") {
1636

1638 double t;
1639 INTPOL_INIT;
1640 INTPOL_3D(t, 1);
1641
1642 /* Calculate molecular density... */
1643 double M = MOLEC_DENS(atm->p[ip], t);
1644
1645 /* Use constant reaction rate... */
1646 double k = GSL_NAN;
1647 if (ctl->oh_chem_reaction == 1)
1648 k = ctl->oh_chem[0];
1649
1650 /* Calculate bimolecular reaction rate... */
1651 else if (ctl->oh_chem_reaction == 2)
1652 k = ctl->oh_chem[0] * exp(-ctl->oh_chem[1] / t);
1653
1654 /* Calculate termolecular reaction rate... */
1655 if (ctl->oh_chem_reaction == 3) {
1656
1657 /* Calculate rate coefficient for X + OH + M -> XOH + M
1658 (JPL Publication 19-05) ... */
1659 double k0 =
1660 ctl->oh_chem[0] * (ctl->oh_chem[1] !=
1661 0 ? pow(298. / t, ctl->oh_chem[1]) : 1.);
1662 double ki =
1663 ctl->oh_chem[2] * (ctl->oh_chem[3] !=
1664 0 ? pow(298. / t, ctl->oh_chem[3]) : 1.);
1665 double c = log10(k0 * M / ki);
1666 k = k0 * M / (1. + k0 * M / ki) * pow(0.6, 1. / (1. + c * c));
1667 }
1668
1669 /* Calculate exponential decay... */
1670 double rate_coef = k * clim_oh(ctl, clim, atm->time[ip], atm->lon[ip],
1671 atm->lat[ip], atm->p[ip]) * M;
1672 double aux = exp(-dt[ip] * rate_coef);
1673 if (ctl->qnt_m >= 0) {
1674 if (ctl->qnt_mloss_oh >= 0)
1675 atm->q[ctl->qnt_mloss_oh][ip]
1676 += atm->q[ctl->qnt_m][ip] * (1 - aux);
1677 atm->q[ctl->qnt_m][ip] *= aux;
1678 }
1679 if (ctl->qnt_vmr >= 0)
1680 atm->q[ctl->qnt_vmr][ip] *= aux;
1681 }

1686void module_h2o2_chem(
1687 ctl_t * ctl,
1688 clim_t * clim,
1689 met_t * met0,
1690 met_t * met1,
1691 atm_t * atm,
1692 double *dt) {
1693
1694 /* Set timer... */
1695 SELECT_TIMER("MODULE_H2O2CHEM", "PHYSICS", NVTX_GPU);
1696
1697 /* Check quantity flags... */
1698 if (ctl->qnt_m < 0 && ctl->qnt_vmr < 0)
1699 ERRMSG("Module needs quantity mass or volume mixing ratio!");
1700 if (ctl->qnt_Cx < 0)
1701 ERRMSG("Module needs quantity implicit volume mixing ratio!");
1702
1703 /* Loop over particles... */
1704 PARTICLE_LOOP(0, atm->np, 1, "acc data present(clim,ctl,met0,met1,atm,dt)") {
1705
1706 /* Check whether particle is inside cloud... */
1707 double lwc;
1708 INTPOL_INIT;
1709 INTPOL_3D(lwc, 1);
1710 if (!(lwc > 0))
1711 continue;
1712
1713 /* Check implicit volume mixing ratio... */
1714 if (atm->q[ctl->qnt_Cx][ip] == 0)
1715 continue;
1716
1717 /* Get temperature... */
1718 double t;
1719 INTPOL_3D(t, 0);
1720
1721 /* Get molecular density... */
1722 double M = MOLEC_DENS(atm->p[ip], t);
1723
1724 /* Reaction rate (Berglen et al., 2004)... */
1725 double k = 9.1e7 * exp(-29700 / RI * (1. / t - 1. / 298.15)); // Maass 1999 unit: M^(-2)
1726
1727 /* Henry constant of SO2... */
1728 double H_SO2 = 1.3e-2 * exp(2900 * (1. / t - 1. / 298.15)) * RI * t;
1729 double K_1S = 1.23e-2 * exp(2.01e3 * (1. / t - 1. / 298.15)); // unit: mol/L
1730
1731 /* Henry constant of H2O2... */
1732 double H_h2o2 = 8.3e2 * exp(7600 * (1 / t - 1 / 298.15)) * RI * t;
1733
1734 /* Concentration of H2O2 (Barth et al., 1989)... */
1735 double SO2 = atm->q[ctl->qnt_Cx][ip] * 1e9; // vmr unit: ppbv
1736 double h2o2 = H_h2o2 * clim_zm(&clim->h2o2, atm->time[ip], atm->lat[ip], atm->p[ip]) * M * 0.59 * exp(-0.687 * SO2) * 1000 / AVO; // unit: mol/L
1737
1738 /* Volume water content in cloud [m^3 m^(-3)]... */
1739 double rho_air = 100 * atm->p[ip] / (RI * t) * MA / 1000;
1740 //MA: Molar mass of dry air; RI: Ideal gas constant 8.314 [J/(mol K)]
1741 double CWC = lwc * rho_air / 1000;
1742
1743 /* Calculate exponential decay (Rolph et al., 1992)... */
1744 double rate_coef = k * K_1S * h2o2 * H_SO2 * CWC;
1745 double aux = exp(-dt[ip] * rate_coef);
1746 if (ctl->qnt_m >= 0) {
1747 if (ctl->qnt_mloss_h2o2 >= 0)
1748 atm->q[ctl->qnt_mloss_h2o2][ip] += atm->q[ctl->qnt_m][ip] * (1 - aux);
1749 atm->q[ctl->qnt_m][ip] *= aux;
1750 }
1751 if (ctl->qnt_vmr >= 0)
1752 atm->q[ctl->qnt_vmr][ip] *= aux;
1753 }
1754}

1980void module_mixing_help(
1981 ctl_t * ctl,
1982 clim_t * clim,
1983 atm_t * atm,
1984 int *ixs,
1985 int *iys,
1986 int *izs,
1987 int qnt_idx) {
1988
1989 double *cmean;
1990
1991 int *count;
1992
1993 /* Allocate... */
1994 ALLOC(cmean, double,
1995 ctl->mixing_nx * ctl->mixing_ny * ctl->mixing_nz);
1996 ALLOC(count, int,
1997 ctl->mixing_nx * ctl->mixing_ny * ctl->mixing_nz);
1998
1999 /* Loop over particles... */
2000 for (int ip = 0; ip < atm->np; ip++)
2001 if (izs[ip] >= 0) {
2002 int idx = ARRAY_3D
2003 (ixs[ip], iys[ip], ctl->mixing_ny, izs[ip], ctl->mixing_nz);

2005 count[idx]++;

2007#pragma omp parallel for
2008 for (int i = 0; i < ctl->mixing_nx * ctl->mixing_ny * ctl->mixing_nz; i++)
2009 if (count[i] > 0)
2010 cmean[i] /= count[i];
2011
2012 /* Calculate interparcel mixing... */
2013#pragma omp parallel for
2014 for (int ip = 0; ip < atm->np; ip++)

2016

2023 }

2026 atm->q[qnt_idx][ip] +=
2027 (cmean
2028 [ARRAY_3D(ixs[ip], iys[ip], ctl->mixing_ny, izs[ip], ctl->mixing_nz)]

2030 }

2033 free(cmean);
2034 free(count);

2365/*****************************************************************************/
2366
2367void module_wet_deposition(
2368 ctl_t * ctl,
2369 met_t * met0,
2370 met_t * met1,
2371 atm_t * atm,
2372 double *dt) {
2373
2374 /* Set timer... */
2375 SELECT_TIMER("MODULE_WETDEPO", "PHYSICS", NVTX_GPU);
2376
2377 /* Check quantity flags... */
2378 if (ctl->qnt_m < 0 && ctl->qnt_vmr < 0)
2379 ERRMSG("Module needs quantity mass or volume mixing ratio!");
2380
2381 /* Loop over particles... */
2382 PARTICLE_LOOP(0, atm->np, 1, "acc data present(ctl,met0,met1,atm,dt)") {
2383
2384 /* Check whether particle is below cloud top... */
2385 double pct;
2386 INTPOL_INIT;
2387 INTPOL_2D(pct, 1);
2388 if (!isfinite(pct) || atm->p[ip] <= pct)
2389 continue;
2390
2391 /* Get cloud bottom pressure... */
2392 double pcb;
2393 INTPOL_2D(pcb, 0);
2394
2395 /* Estimate precipitation rate (Pisso et al., 2019)... */
2396 double cl;
2397 INTPOL_2D(cl, 0);
2398 double Is =
2399 pow(1. / ctl->wet_depo_pre[0] * cl, 1. / ctl->wet_depo_pre[1]);
2400 if (Is < 0.01)
2401 continue;
2402
2403 /* Check whether particle is inside or below cloud... */
2404 double iwc, lwc;
2405 INTPOL_3D(lwc, 1);
2406 INTPOL_3D(iwc, 0);
2407 int inside = (iwc > 0 || lwc > 0);
2408
2409 /* Get temperature... */
2410 double t;
2411 INTPOL_3D(t, 0);
2412
2413 /* Calculate in-cloud scavenging coefficient... */
2414 double lambda = 0;
2415 if (inside) {
2416
2417 /* Calculate retention factor... */
2418 double eta;
2419 if (t > 273.15)
2420 eta = 1;
2421 else if (t <= 238.15)
2422 eta = ctl->wet_depo_ic_ret_ratio;
2423 else
2424 eta = LIN(273.15, 1, 238.15, ctl->wet_depo_ic_ret_ratio, t);
2425
2426 /* Use exponential dependency for particles ... */
2427 if (ctl->wet_depo_ic_a > 0)
2428 lambda = ctl->wet_depo_ic_a * pow(Is, ctl->wet_depo_ic_b) * eta;
2429
2430 /* Use Henry's law for gases... */
2431 else if (ctl->wet_depo_ic_h[0] > 0) {
2432
2433 /* Get Henry's constant (Sander, 2015)... */
2434 double h = ctl->wet_depo_ic_h[0]
2435 * exp(ctl->wet_depo_ic_h[1] * (1. / t - 1. / 298.15));
2436
2437 /* Use effective Henry's constant for SO2
2438 (Berglen, 2004; Simpson, 2012)... */
2439 if (ctl->wet_depo_ic_h[2] > 0) {
2440 double H_ion = pow(10, ctl->wet_depo_ic_h[2] * (-1));
2441 double K_1 = 1.23e-2 * exp(2.01e3 * (1. / t - 1. / 298.15));
2442 double K_2 = 6e-8 * exp(1.12e3 * (1. / t - 1. / 298.15));
2443 h *= (1 + K_1 / H_ion + K_1 * K_2 / pow(H_ion, 2));
2444 }
2445
2446 /* Estimate depth of cloud layer... */
2447 double dz = 1e3 * (Z(pct) - Z(pcb));
2448
2449 /* Calculate scavenging coefficient (Draxler and Hess, 1997)... */
2450 lambda = h * RI * t * Is / 3.6e6 / dz * eta;
2451 }
2452 }
2453
2454 /* Calculate below-cloud scavenging coefficient... */
2455 else {
2456
2457 /* Calculate retention factor... */
2458 double eta;
2459 if (t > 270)
2460 eta = 1;
2461 else
2462 eta = ctl->wet_depo_bc_ret_ratio;
2463
2464 /* Use exponential dependency for particles... */
2465 if (ctl->wet_depo_bc_a > 0)
2466 lambda = ctl->wet_depo_bc_a * pow(Is, ctl->wet_depo_bc_b) * eta;
2467
2468 /* Use Henry's law for gases... */
2469 else if (ctl->wet_depo_bc_h[0] > 0) {
2470
2471 /* Get Henry's constant (Sander, 2015)... */
2472 double h = ctl->wet_depo_bc_h[0]
2473 * exp(ctl->wet_depo_bc_h[1] * (1. / t - 1. / 298.15));
2474
2475 /* Estimate depth of cloud layer... */
2476 double dz = 1e3 * (Z(pct) - Z(pcb));
2477
2478 /* Calculate scavenging coefficient (Draxler and Hess, 1997)... */
2479 lambda = h * RI * t * Is / 3.6e6 / dz * eta;
2480 }
2481 }
2482
2483 /* Calculate exponential decay of mass... */
2484 double aux = exp(-dt[ip] * lambda);
2485 if (ctl->qnt_m >= 0) {
2486 if (ctl->qnt_mloss_wet >= 0)
2487 atm->q[ctl->qnt_mloss_wet][ip]
2488 += atm->q[ctl->qnt_m][ip] * (1 - aux);
2489 atm->q[ctl->qnt_m][ip] *= aux;
2490 }
2491 if (ctl->qnt_vmr >= 0)
2492 atm->q[ctl->qnt_vmr][ip] *= aux;
2493 }
2494}
